DROP TABLE CS_ATTACHMENT;
DROP SEQUENCE SEQ_CS_ATTACHMENT;
CREATE SEQUENCE SEQ_CS_ATTACHMENT NOCACHE;

CREATE TABLE CS_ATTACHMENT (
	FILE_NO NUMBER PRIMARY KEY,
	ORIGIN_NAME VARCHAR2(100) NOT NULL,
	CHANGE_NAME VARCHAR2(100) NOT NULL,
	FILE_PATH VARCHAR2(1000) NOT NULL,
	CREATE_DATE DATE DEFAULT SYSDATE,
    STATUS VARCHAR2(1) DEFAULT 'Y' CHECK (STATUS IN('Y', 'N')),
	REF_INO NUMBER NOT NULL,
    FOREIGN KEY (REF_INO) REFERENCES INQUIRY (I_NO)
);

COMMENT ON COLUMN CS_ATTACHMENT.FILE_NO IS '파일번호';
COMMENT ON COLUMN CS_ATTACHMENT.ORIGIN_NAME IS '원본파일명';
COMMENT ON COLUMN CS_ATTACHMENT.CHANGE_NAME IS '수정파일명';
COMMENT ON COLUMN CS_ATTACHMENT.FILE_PATH IS '파일경로';
COMMENT ON COLUMN CS_ATTACHMENT.CREATE_DATE IS '파일업로드날짜';
COMMENT ON COLUMN CS_ATTACHMENT.STATUS IS '파일삭제여부 유지중=Y / 삭제됨=N';
COMMENT ON COLUMN CS_ATTACHMENT.REF_INO IS '참조1:1문의번호';
